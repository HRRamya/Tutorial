ROLE:-

CREATE TABLE ROLE (
ROLE_ID VARCHAR(25) PRIMARY KEY, 
ROLE_NAME VARCHAR(25)  NOT NULL, 
USERNAME VARCHAR(25) NOT NULL, 
FOREIGN KEY(USERNAME) REFERENCES USER(EMAILID)
);
USER:-

CREATE TABLE USER (USERID VARCHAR2(10) PRIMARY KEY, USERNAME VARCHAR2(25) NOT NULL,  ADDRESS VARCHAR2(50),EMAILID VARCHAR2(25), MOBILE VARCHAR2(10) NOT NULL, PASSWORD VARCHAR2(10), ROLENAME VARCHAR(25) NOT NULL ,FOREIGN KEY (ROLENAME) REFERENCES ROLE(ROLE_NAME));


CATEGORY:-

CREATE TABLE CATEGORY (C_ID VARCHAR(15) PRIMARY KEY, C_NAME VARCHAR(25) NOT NULL,  C_DESCRIPTION VARCHAR(50));

SUPPLIER:-

CREATE TABLE SUPPLIER (S_ID VARCHAR(15) PRIMARY KEY, S_NAME VARCHAR(25) NOT NULL,  S_ADDRESS VARCHAR(50), S_PHONENO NUMBER(10) NOT NULL);

PRODUCT:-

CREATE TABLE PRODUCT (P_ID VARCHAR(15) PRIMARY KEY, P_NAME VARCHAR(50) NOT NULL, P_PRICE NUMBER(25) NOT NULL, P_QUANTITY NUMBER(25) NOT NULL, P_DESCRIPTION VARCHAR(25), C_ID VARCHAR(25) NOT NULL, S_ID VARCHAR(25) NOT NULL, FOREIGN KEY (C_ID) REFERENCES CATEGORY(C_ID), FOREIGN KEY (S_ID) REFERENCES SUPPLIER(S_ID));


CART:-

CREATE TABLE CART(CART_ID VARCHAR(20) PRIMARY KEY, CUSTOMER_ID VARCHAR(20) NOT NULL, GRANDTOTAL VARCHAR(20), FOREIGN KEY (CUSTOMER_ID) REFERENCES CUSTOMER(CUSTOMER_ID));
ALTER TABLE CART DROP COLUMN CUSTOMER_ID;
ALTER TABLE CART ADD COLUMN USERID VARCHAR(10);
ALTER TABLE CART ADD FOREIGN KEY(USERID) REFERENCES USER(USERID);




CARTITEMS:-

CREATE TABLE CARTITEMS(CARTITEMS_ID VARCHAR(20) PRIMARY KEY, CART_ID VARCHAR(20) NOT NULL, CUSTOMER_ID VARCHAR(20) NOT NULL, P_ID VARCHAR2(15) NOT NULL, QUANTITY INT(20) NOT NULL, TOTALPRICE VARCHAR(20) NOT NULL, FOREIGN KEY (CUSTOMER_ID) REFERENCES CUSTOMER(CUSTOMER_ID), FOREIGN KEY (CART_ID) REFERENCES CART(CART_ID), FOREIGN KEY (P_ID) REFERENCES PRODUCT(P_ID));
ALTER TABLE CARTITEMS DROP COLUMN CUSTOMER_ID;
ALTER TABLE CARTITEMS ADD COLUMN USERID VARCHAR(10);
ALTER TABLE CARTITEMS ADD FOREIGN KEY(USERID) REFERENCES USER(USERID);
ALTER TABLE CARTITEMS ADD COLUMN P_NAME VARCHAR(50);
ALTER TABLE CARTITEMS ADD FOREIGN KEY(P_NAME) REFERENCES PRODUCT(P_NAME);


ORDERS:-

create table orders(order_id varchar(20) primary key, userid varchar(10) not null, p_id varchar(15), b_address varchar(20) not null, s_address varchar(20) not null, totalprice varchar(20) not null, foreign key(userid) references user(userid), foreign key(p_id) references product(p_id));
alter table orders add column payment varchar(20) not null;


ORDERLIST:-

CREATE TABLE ORDERLIST(ORDERLIST_ID VARCHAR2(20) PRIMARY KEY, ORDER_ID VARCHAR2(20) NOT NULL, P_ID VARCHAR2(15) NOT NULL, FOREIGN KEY (ORDER_ID) REFERENCES ORDERS(ORDER_ID), FOREIGN KEY (P_ID) REFERENCES PRODUCT(P_ID));


ADDRESS:-

CREATE TABLE ADDRESS(ADDRESS_ID VARCHAR2(20) PRIMARY KEY, ORDER_ID VARCHAR2(20) NOT NULL, ORDERLIST_ID VARCHAR2(15) NOT NULL, FOREIGN KEY (ORDER_ID) REFERENCES ORDERS(ORDER_ID), FOREIGN KEY (ORDERLIST_ID) REFERENCES ORDERLIST(ORDERLIST_ID));


PAYMENT:-

CREATE TABLE PAYMENT(PAYMENT_ID VARCHAR2(20) PRIMARY KEY, ORDER_ID VARCHAR2(20) NOT NULL, ORDERLIST_ID VARCHAR2(15) NOT NULL, TOTALAMOUNT VARCHAR2(20), DELIVERYDATE DATE, FOREIGN KEY (ORDER_ID) REFERENCES ORDERS(ORDER_ID), FOREIGN KEY (ORDERLIST_ID) REFERENCES ORDERLIST(ORDERLIST_ID));


ONLINEPAYMENT:-

CREATE TABLE ONLINEPAYMENT(ONLINE_ID VARCHAR2(20) PRIMARY KEY, USERID VARCHAR2(10) NOT NULL,CARDNAME VARCHAR2(25) NOT NULL, CARDNUMBER VARCHAR2(20), EXPIRYDATE DATE, FOREIGN KEY (USERID) REFERENCES USER(USERID));


BILLING:

CREATE TABLE BILLING(BILLING_ID VARCHAR(20) PRIMARY KEY, CUSTOMER_ID VARCHAR(10) NOT NULL, ADDRESS VARCHAR(20), FOREIGN KEY(CUSTOMER_ID) REFERENCES CUSTOMER(CUSTOMER_ID));
alter table billing drop column address;
alter table billing add column b_address varchar(20) not null;

SHIPPING:

CREATE TABLE SHIPPING(SHIPPING_ID VARCHAR(20) PRIMARY KEY, CUSTOMER_ID VARCHAR(20) NOT NULL, ADDRESS VARCHAR(20), FOREIGN KEY(CUSTOMER_ID) REFERENCES CUSTOMER(CUSTOMER_ID));
alter table shipping drop column address;
alter table shipping add column s_address varchar(20) not null;


CUSTOMER:

CREATE TABLE CUSTOMER(
CUSTOMER_ID VARCHAR(20) PRIMARY KEY, 
CUSTOMER_NAME VARCHAR(25) NOT NULL, 
CUSTOMER_MOBILE VARCHAR(10) NOT NULL, 
CUSTOMER_EMAILID VARCHAR(25) NOT NULL,
BILLING_ID VARCHAR(20), 
SHIPPING_ID VARCHAR(20),
USERID VARCHAR(10) NOT NULL,
CART_ID VARCHAR(20), 
FOREIGN KEY(USERID) REFERENCES USER(USERID), 
FOREIGN KEY(BILLING_ID) REFERENCES BILLING(BILLING_ID), 
FOREIGN KEY(SHIPPING_ID) REFERENCES SHIPPING(SHIPPING_ID),
FOREIGN KEY(CART_ID) REFERENCES CART(CART_ID));
